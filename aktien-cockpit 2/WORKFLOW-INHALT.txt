name: Ticker-Datenbank aktualisieren

on:
  schedule:
    - cron: "30 4 * * 1"   # jeden Montag 04:30 UTC
  workflow_dispatch: {}     # manueller Start ueber den Actions-Tab

permissions:
  contents: write

jobs:
  build:
    runs-on: ubuntu-latest
    steps:
      - uses: actions/checkout@v4

      - uses: actions/setup-python@v5
        with:
          python-version: "3.12"

      - name: Ticker-Datenbank neu erzeugen
        run: python3 scripts/build_tickers.py

      - name: Aenderungen committen
        run: |
          git config user.name "ticker-bot"
          git config user.email "actions@users.noreply.github.com"
          git add data/tickers.json
          git diff --cached --quiet && echo "Keine Aenderungen." && exit 0
          git commit -m "Ticker-Datenbank aktualisiert ($(date -u +%F))"
          git push
