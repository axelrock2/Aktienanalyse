name: Nachrichten aktualisieren

on:
  schedule:
    - cron: "15 * * * *"    # stuendlich zur 15. Minute (UTC)
  workflow_dispatch: {}      # manueller Start ueber den Actions-Tab

permissions:
  contents: write

concurrency:
  group: news
  cancel-in-progress: false

jobs:
  build:
    runs-on: ubuntu-latest
    steps:
      - uses: actions/checkout@v4

      - uses: actions/setup-python@v5
        with:
          python-version: "3.12"

      - name: Nachrichten sammeln
        run: python3 scripts/build_news.py

      - name: Aenderungen committen
        run: |
          git config user.name "news-bot"
          git config user.email "actions@users.noreply.github.com"
          git add data/news.json
          git diff --cached --quiet && echo "Keine Aenderungen." && exit 0
          git commit -m "Nachrichten aktualisiert ($(date -u '+%F %H:%M UTC'))"
          git pull --rebase --autostash || true
          git push
